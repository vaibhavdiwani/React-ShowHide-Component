## Install

```bash
npm install --save react-showhide-component
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-showhide-component'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
